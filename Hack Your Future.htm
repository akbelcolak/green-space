Debugger

Use a browser debugger to find the mistake in the code below.


const dividend = 18;
const numbers = [2, 3, 4, 5, 6, 7, 8, 9];
let idx;
let factor;
let divisor;

for (idx = 0; idx < numbers.length; idx++) {
  divisor = numbers[idx];

  if (dividend % divisor === 0) {
    factor = true;
  }

  if (factor) {
    console.log(divisor + ' is a factor of ' + dividend + '!');
  }
}
